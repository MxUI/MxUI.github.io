67  //- Filter based on single time value
68  template<typename TYPE>
69  TYPE filter( time_type focus, const std::vector<std::pair<std::pair<time_type,time_type>, TYPE> > &points ) const {
70  TYPE sum = TYPE(0);
71 
72  for( auto i: points ) {
73  if ( i.first.first <= focus + right && i.first.first >= focus - left ) {
74  sum += i.second;
75  }
76  }
77 
78  if ( points.size() )
79  return sum / TYPE(points.size());
80  else
81  return TYPE(0);
82  }

87  TYPE sum = TYPE(0);

91  i.first.second <= focus.second + right && i.first.second >= focus.second - left ) {
92  sum += i.second;
93  }
94  }
95

97  return sum / TYPE(points.size());
98  else
99  return TYPE(0);
100  }
