100  void swap( spatial_storage& rhs ) noexcept(noexcept(BIN(std::move(rhs.bin_)))) {
101  data_.swap(rhs.data_);
102  if( is_bin_ && rhs.is_bin_ ) bin_.swap(rhs.bin_);
103  else if( is_bin_ && !rhs.is_bin_ ) {
104  ::new(&(rhs.bin_)) BIN(std::move(bin_));
105  rhs.is_bin_ = true;
106  destroy_if_bin_();
107  } else if( !is_bin_ && rhs.is_bin_ ) {
108  ::new(&(bin_)) BIN(std::move(rhs.bin_));
109  is_bin_ = true;

116  template<typename REGION, typename FOCUS, typename SAMPLER, typename ... ADDITIONAL>
117  typename SAMPLER::OTYPE
118  query(const REGION& reg, const FOCUS& f, SAMPLER& s, ADDITIONAL && ... additional) const {
119  using vec = std::vector<std::pair<point_type,typename SAMPLER::ITYPE> >;
120  if( data_.empty() )
121  return s.filter( f, virtual_container<typename SAMPLER::ITYPE,CONFIG>(vec(),std::vector<bool>()), additional... );
122  if( !is_built() ) EXCEPTION(std::logic_error("MUI Error [spatial_storage.h]: Query error. "
123  "Bin not built yet. Internal data corrupted."));
124 
125  const vec& st = storage_cast<const vec&>(data_);
126 
127  return s.filter( f, virtual_container<typename SAMPLER::ITYPE,CONFIG>(st,bin_.query(reg)), additional...);

150  void insert( storage_t storage ) {
151  destroy_if_bin_();
152  if( !storage ) return;
153  if( !data_ ) data_ = std::move(storage);
154  else if( data_.which() == storage.which() ) data_.apply_visitor(insert_{storage});
155  else EXCEPTION(bad_storage_id("MUI Error [spatial_storage.h]: Insert error. Type doesn't match."));
