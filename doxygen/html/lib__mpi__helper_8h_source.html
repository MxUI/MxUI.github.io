68 template<typename T> inline std::vector<T> gather( T t, MPI_Comm comm ) {
69  int size, rank;
70  MPI_Comm_size( comm, &size );
71  MPI_Comm_rank( comm, &rank );
72  std::vector<T> v;
73  if ( rank == 0 ) v.resize( size );
74  MPI_Gather( &t, 1, mpi_type(t), v.data(), 1, mpi_type(t), 0, comm );
75  return v;
76 }
