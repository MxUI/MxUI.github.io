67  //- Filter based on single time value
68  template<typename TYPE>
69  TYPE filter( time_type focus, const std::vector<std::pair<std::pair<time_type,time_type>, TYPE> > &points ) const {
70  TYPE sum = TYPE(0);
71 
72  for( auto i: points ) {
73  if ( i.first.first <= focus + right && i.first.first >= focus - left ) {
74  sum += i.second;
75  }
76  }
77 
78  return sum;
79  }

84  TYPE sum = TYPE(0);

88  i.first.second <= focus.second + right && i.first.second >= focus.second - left) {
89  sum += i.second;
90  }
91  }
92 
93  return sum;
94  }
