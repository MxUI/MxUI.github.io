67  //- Filter based on single time value
68  template<typename TYPE>
69  TYPE filter( time_type focus, const std::vector<std::pair<std::pair<time_type,time_type>, TYPE> > &points ) const {
70  REAL wsum = REAL(0);
71  TYPE vsum = TYPE(0);
72  for( auto i: points ) {
73  time_type dt = std::abs(i.first.first - focus);
74  if ( dt < cutoff ) {
75  REAL w = pow( 2*PI*sigma, -0.5 ) * exp( -0.5 * dt * dt / sigma );
76  vsum += i.second * w;
77  wsum += w;
78  }
79  }
80  return ( wsum > std::numeric_limits<REAL>::epsilon() ) ? ( vsum / wsum ) : TYPE(0);
81  }
82 
83  //- Filter based on two time values
84  template<typename TYPE>
85  TYPE filter( std::pair<time_type,time_type> focus, const std::vector<std::pair<std::pair<time_type,time_type>, TYPE> > &points ) const {
86  REAL wsum = REAL(0);
87  TYPE vsum = TYPE(0);
88  for( auto i: points ) {
89  time_type dt1 = std::abs(i.first.first - focus.first);
90  time_type dt2 = std::abs(i.first.second - focus.second);
91  if ( dt1 < cutoff && dt2 < cutoff ) {
92  REAL w = pow( 2*PI*sigma, -0.5 ) * exp( -0.5 * dt1 * dt1 / sigma );
93  vsum += i.second * w;
94  wsum += w;
95  }
96  }
97  return ( wsum > std::numeric_limits<REAL>::epsilon() ) ? ( vsum / wsum ) : TYPE(0);
98  }
99 
100  time_type get_upper_bound( time_type focus ) const {

102  }
103 
104  time_type get_lower_bound( time_type focus ) const {
105  return focus - cutoff;
106  }
107 
108  time_type tolerance() const {
109  return time_type(0);
110  }

112 protected:
113  time_type cutoff;
114  REAL sigma;
