72  template<typename... types>
73  static message make( const id_type& id, types&&... data ) {
74  message msg;
75  msg.id_ = id;
76  msg.id_size_ = streamed_size(id);
77  std::size_t n = msg.id_size_ + streamed_size(data...);
78  msg.data_.resize(n);
79  auto stream = make_ostream(msg.data_.data());
80  stream << id << std::forward_as_tuple(data...);
81  return msg;
82  }
83  static message make( std::vector<char> data ){
84  message m;
85  m.data_.swap(data);
86  auto in = make_istream(m.data_.begin());
87  in >> m.id_;
88  m.id_size_ = streamed_size(m.id_);
89  return m;
90  }
