69  sampler_sum_quintic( REAL r_ ) : r( r_ ), hinv( REAL( 3 ) / r_ )
70  {
71  static_assert( D == 1 || D == 2 || D == 3, "Quintic kernel for dimension other than 1,2,3 not defined." );
72  REAL sigma;
73  switch( D ) {
74  case 1:
75  sigma = 1.0 / 120.0;
76  break;
77  case 2:
78  sigma = 7.0 / 478.0 / PI;
79  break;
80  case 3:
81  sigma = 1.0 / 120.0 / PI;
82  break;
83  }
84  norm_factor = sigma * powr<D>( hinv );
85  }

87  template<template<typename, typename> class CONTAINER>
88  inline OTYPE filter( point_type focus, const CONTAINER<ITYPE, CONFIG> &data_points ) const
89  {
90  OTYPE vsum = 0;
91  for( size_t i = 0 ; i < data_points.size() ; i++ ) {
92  auto dist2 = normsq( focus - data_points[i].first );
93  if( dist2 < r * r ) {
94  REAL w = quintic_polynomial( sqrt( dist2 ) );
95  vsum += data_points[i].second * w;
96  }
97  }
98  return vsum;
99  }

101  inline geometry::any_shape<CONFIG> support( point_type focus, REAL domain_mag ) const
102  {

104  }

109  inline REAL quintic_polynomial( const REAL dist ) const
110  {
111  REAL s, w;
112  REAL s1, s2, s3;
113  REAL s1_5, s2_5, s3_5;
114  s = dist * hinv;
115  s1 = 1.0 - s;
116  s2 = 2.0 - s;
117  s3 = 3.0 - s;
118  s1_5 = 15.0 * powr<5>( s1 );
119  s2_5 = -6.0 * powr<5>( s2 );
120  s3_5 = powr<5>( s3 );
121  if( s < 1.0 ) {
122  w = s3_5 + s2_5 + s1_5;
123  } else if( s < 2.0 ) {
124  w = s3_5 + s2_5;
125  } else if( s < 3.0 ) {
126  w = s3_5;
127  } else {
128  w = 0.0;
129  }
130  w *= norm_factor;
131  return w;
132  }
