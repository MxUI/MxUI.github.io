176  bool scan_spans_(time_type t, const span_t& s, const spans_type& spans ) const {
177  auto end = spans.upper_bound(std::make_pair(t,t));
178  bool prefetched = false;
179 
180  for( auto itr = spans.begin(); itr != end; ++itr ) {
181  if( t < itr->first.second || almost_equal(t, itr->first.second) ){
182  prefetched = true;
183  if( collide(s,itr->second) ) return true;
184  }
185  }
186  // if prefetched at t, but no overlap region, then return false;
187  // otherwise return true;
188  return !prefetched;
189  }

263  template<typename TYPE>
264  void push( const std::string& attr, const point_type& loc, const TYPE& value ) {
265  if( FIXEDPOINTS ) {
266  // If this push is before first commit then build local points list
267  if( !initialized_pts_ ) push_buffer_pts.emplace_back( loc );
268 
269  storage_raw_t& n = push_buffer_raw[attr];
270  if( !n ) n = storage_raw_t(std::vector<std::pair<size_t,TYPE> >());
271  storage_cast<std::vector<std::pair<size_t,TYPE> >&>(n).emplace_back( fixedPointCount_, value );
272 
273  // Increment counter for flat fixed point list
274  fixedPointCount_++;
275  }
276  else {
277  storage_t& n = push_buffer[attr];
278  if( !n ) n = storage_t(std::vector<std::pair<point_type,TYPE> >());
279  storage_cast<std::vector<std::pair<point_type,TYPE> >&>(n).emplace_back( loc, value );
280  }
281  }

537  int commit( time_type t1, time_type t2 = std::numeric_limits<time_type>::lowest() ) {
538  std::vector<bool> is_sending(comm->remote_size(), true);
539  std::pair<time_type,time_type> time(t1,t2);
540 
541  // Check if peer set to disabled (not linked to time span)
542  for( std::size_t i=0; i<peers.size(); ++i ) {
543  if(peers[i].is_recv_disabled())
544  is_sending[i] = false;
545  }
546 
547  // Check for smart send based on t1
548  if( (((span_start < t1) || almost_equal(span_start, t1)) &&
549  ((t1 < span_timeout) || almost_equal(t1, span_timeout))) ) {
