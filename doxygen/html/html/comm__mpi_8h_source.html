74  void init( const char URI[], MPI_Comm world ) {
75  if ( initialized )
76  throw( std::runtime_error("MUI Error [comm_mpi.h]: Duplicate MUI communicator initialization") );
77 
78  // check MPI initialization status
79  int init;
80  MPI_Initialized( &init );
81  if (!init) {
82  init_by_me = true;
83  MPI_Init( NULL, NULL );
84  }
85 
86  // duplicate world to avoid potential conflict with host program
87  if (world == MPI_COMM_WORLD) MPI_Comm_dup( MPI_COMM_WORLD, &world );
88  MPI_Comm_size( world, &global_size_ );
89  MPI_Comm_rank( world, &global_rank_ );
90  if (global_size_ < 2)
91  std::cout << "MUI Warning [comm_mpi.h]: Number of global ranks less than 2" << std::endl;
92 
93  // get upper bond for tag hash
94  int prime;
95  {
96  int* v;
97  int flag;
98  MPI_Comm_get_attr(MPI_COMM_WORLD,MPI_TAG_UB,&v,&flag);
99  prime = *v;
100  }
101 
102  // parse URI, split the world using domain tag
103  uri desc(URI);
104  uri_host_ = desc.host();
105  uri_path_ = desc.path();
106  uri_protocol_ = desc.protocol();
107  int domain_hash = std::hash<std::string>()( desc.host() );
108  domain_hash=std::abs(domain_hash);
109  int ifs_hash = std::hash<std::string>()( desc.path() );
110  ifs_hash=std::abs(ifs_hash);
111  MPI_Comm_split( world, domain_hash % prime, global_rank_, &domain_local_ );
112  MPI_Comm_size( domain_local_, &local_size_ );
113  MPI_Comm_rank( domain_local_, &local_rank_ );
114 
115  // set intercommunication using interface tag
116  std::vector<int> all_domain( global_size_, 0 );
117  std::vector<int> all_ifs ( global_size_, 0 );
118  MPI_Allgather( &domain_hash, 1, MPI_INT, &all_domain[0], 1, MPI_INT, world );
119  MPI_Allgather( &ifs_hash, 1, MPI_INT, &all_ifs[0], 1, MPI_INT, world );
120  for( int i = 0 ; i < global_size_ ; i++ ) {
121  if ( i == global_rank_ ) continue;
122  if ( all_domain[i] != domain_hash && all_ifs[i] == ifs_hash ) {
123  MPI_Intercomm_create( domain_local_, 0, world, i, ifs_hash % prime, &domain_remote_ );
124  break;
125  }
126  }
127  MPI_Comm_remote_size( domain_remote_, &remote_size_ );
128 
129  // output for debugging
130  for( int i = 0; i < global_size_; i++ ) {
131  if ( i == global_rank_ ) {
132  std::cout << "MUI [comm_mpi.h]: rank " << global_rank_ << '\t'
133  << "identifier " << URI << '\t'
134  << "domain size " << local_size_ << '\t'
135  << "peer number " << remote_size_ << '\t'
136  << std::endl;
137  }
138  MPI_Barrier( world );
139  }
140 
141  initialized = true;
142  }
143 
144  void finalize() {
145  if ( initialized && init_by_me ) MPI_Finalize();
146  }
