67  parts.push_back(s);
68  return *this;

70  void operator && ( const geometry::any_shape<CONFIG> &s ) {
71  auto i = parts.begin();
72  while( i != parts.end() ) {
73  if ( !i->collide(s) ) i = parts.erase(i);
74  else ++i;
75  }
76  }
77  void reset() { parts.clear(); }
