62 inline std::vector<std::unique_ptr<uniface<CONFIG>>> create_uniface( std::string domain, std::vector<std::string> interfaces, MPI_Comm world = MPI_COMM_WORLD )
63 {
64  // gather total number of interfaces (edges of the graph)
65  int global_size, global_rank;
66  MPI_Comm_size( world, &global_size );
67  MPI_Comm_rank( world, &global_rank );
68 
69  std::map<int, std::string> map;
70  std::vector<int> my_hashes;
71  for( auto &i : interfaces ) {
72  auto h = std::hash<std::string>()( i );
73  my_hashes.push_back( h );
74  map[h] = i;
75  }
76  for( int i = 0; i < global_size; i++ ) {
77  if( global_rank == i ) {
78  for( auto &e : map ) {
79  std::cout << "> rank " << global_rank << " \"" << domain.c_str()
80  << "\" registered interface " << e.second.c_str()
81  << " as " << std::hex << e.first << std::dec << std::endl;
82  }
83  }
84  MPI_Barrier( world );
85  }
86 
87  int n_unique;
88  std::set<int> unique_ifs;
89  if( global_rank == 0 ) {
90  std::vector<int> nifs = gather( ( int )interfaces.size(), world );
91  std::vector<int> displs( global_size + 1, 0 );
92  std::partial_sum( nifs.begin(), nifs.end(), displs.begin() + 1 );
93 
94  std::vector<int> all_hashes( displs.back() );
95  MPI_Gatherv( my_hashes.data(), my_hashes.size(), MPI_INT, all_hashes.data(), nifs.data(), displs.data(), MPI_INT, 0, world );
96 
97  for( auto &i : all_hashes ) unique_ifs.insert( i );
98  n_unique = unique_ifs.size();
99  std::cout << "> " << n_unique << " distinct interfaces found" << std::endl;
100  } else {
101  gather( ( int )interfaces.size(), world );
102  MPI_Gatherv( my_hashes.data(), my_hashes.size(), MPI_INT, NULL, NULL, NULL, MPI_INT, 0, world );
103  }
104  MPI_Barrier( world );
105  MPI_Bcast( &n_unique, 1, MPI_INT, 0, world );
106  std::vector<int> uniq_hashes( n_unique );
107  if( global_rank == 0 ) uniq_hashes.assign( unique_ifs.begin(), unique_ifs.end() );
108  MPI_Bcast( uniq_hashes.data(), n_unique, MPI_INT, 0, world );
109 
110  std::vector<std::unique_ptr<uniface<CONFIG>>> unifaces;
111  for( auto &i : uniq_hashes ) {
112  MPI_Comm comm_ifs;
113  if( map.find( i ) != map.end() ) {
114  MPI_Comm_split( world, 1, global_rank, &comm_ifs );
115  int comm_rank;
116  MPI_Comm_rank( comm_ifs, &comm_rank );
117  if( comm_rank == 0 ){
118  std::cout << "> setting up interface " << map[i].c_str() << " [" << std::hex << i
119  << std::dec << "] (rank ids are local to each interface)" << std::endl;
120  }
121  std::string full_uri( "mpi://" );
122  full_uri = full_uri + domain + "/" + map[i];
123  unifaces.push_back( std::unique_ptr<uniface<CONFIG>>(new uniface<CONFIG>( new comm_mpi_smart( full_uri.c_str(), comm_ifs ) )) );
124  } else {
125  MPI_Comm_split( world, 0, global_rank, &comm_ifs );
126  }
127 
128  MPI_Barrier( world );
129  }
130 
131  return unifaces;
132 }

134  // Ensure all interfaces in the range of the two iterators have hit sync_all
135  template<typename IteratorT>
136  void sync_all(IteratorT begin, IteratorT end,
137  typename std::decay<decltype(**begin)>::type::time_type t) {
138  for(auto it = begin; it != end; ++it) {
139  auto& unif = *it;
140  // If nothing pushed this just announces the current time
141  unif->commit(t);
142  }
143 
144  for(auto it = begin; it != end; ++it) {
145  auto& unif = *it;
146  // This waits for all the other end of the interface to announce
147  // a time greater than or equal to t
148  unif->barrier(t);
149  }
150  }
