70  template<template<typename,typename> class CONTAINER>
71  inline OTYPE filter( point_type focus, const CONTAINER<ITYPE,CONFIG> &data_points ) const {
72  size_t n(0);
73  OTYPE vsum(0);
74  for( size_t i = 0 ; i < data_points.size() ; i++ ) {
75  point_type dx(REAL(0.0));
76  for (INT j = 0 ; j < CONFIG::D ; j++) {
77  dx[j] = std::fabs(data_points[i].first[j] - focus[j]);
78  }
79  bool within = true;
80  for( INT i = 0 ; within && i < CONFIG::D ; i++ ) within = within && ( dx[i] < bbox[i] );
81  if ( within ) {
82  vsum += data_points[i].second;
83  n++;
84  }
85  }
86  if (CONFIG::DEBUG) assert( n!=0 );
87  return n ? ( vsum / OTYPE(n) ): OTYPE(0.);
88  }
